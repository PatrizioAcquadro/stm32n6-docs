90 def _parse_preprocessing_section(cfg: DictConfig, mode:str = None) -> None:
91  # cfg: 'preprocessing' section of the configuration file
92  legal = ["rescaling", "resizing", "color_mode"]
93  if mode == 'deployment':
94  # removing the obligation to have rescaling for the 'deployment' mode
95  required=["resizing", "color_mode"]
96  check_config_attributes(cfg.preprocessing, specs={"legal": legal, "all": required}, section="preprocessing")
97  else:
98  required=legal
99  check_config_attributes(cfg.preprocessing, specs={"legal": legal, "all": required}, section="preprocessing")
100  legal = ["scale", "offset"]
101  check_config_attributes(cfg.preprocessing.rescaling, specs={"legal": legal, "all": legal}, section="preprocessing.rescaling")
102 
103  legal = ["interpolation", "aspect_ratio"]
104  check_config_attributes(cfg.preprocessing.resizing, specs={"legal": legal, "all": legal}, section="preprocessing.resizing")
105 
106  if cfg.hardware_type == "MCU":
107  if cfg.preprocessing.resizing.aspect_ratio not in ("fit", "crop", "padding"):
108  raise ValueError("\nSupported methods for resizing images are 'fit', 'crop' and 'padding'. "
109  f"Received {cfg.preprocessing.resizing.aspect_ratio}\n"
110  "Please check the `resizing.aspect_ratio` attribute in "
111  "the 'preprocessing' section of your configuration file.")
112 
113  elif cfg.hardware_type == "MPU":
114  if cfg.preprocessing.resizing.aspect_ratio not in ["fit","padding"]:
115  raise ValueError("The only values of aspect_ratio that are supported at this point are 'fit' and 'padding'"
116  "('crop' is not supported).")
117  else:
118  raise ValueError(f"Unsupported hardware_type: {cfg.hardware_type}. "
119  "Expected 'MCU' or 'MPU'.")
120 
121  # Check resizing interpolation value
122  interpolation_methods = ["bilinear", "nearest", "area", "lanczos3", "lanczos5", "bicubic", "gaussian",
123  "mitchellcubic"]
124  if cfg.preprocessing.resizing.interpolation not in interpolation_methods:
125  raise ValueError(f"\nUnknown value for `interpolation` attribute. Received {cfg.preprocessing.resizing.interpolation}\n"
126  f"Supported values: {interpolation_methods}\n"
127  "Please check the 'preprocessing.resizing' section of your configuration file.")
128 
129  # Check color mode value
130  color_modes = ["grayscale", "rgb", "rgba"]
131  if cfg.preprocessing.color_mode not in color_modes:
132  raise ValueError(f"\nUnknown value for `color_mode` attribute. Received {cfg.preprocessing.color_mode}\n"
133  f"Supported values: {color_modes}\n"
134  "Please check the 'preprocessing' section of your configuration file.")

137 def _parse_data_augmentation_section(cfg: DictConfig) -> None:
138  """
139  This function checks the data augmentation section of the config file.
140  The attribute that introduces the section is either `data_augmentation`
141  or `custom_data_augmentation`. If it is `custom_data_augmentation`,
142  the name of the data augmentation function that is provided must be
143  different from `data_augmentation` as this is a reserved name.
144 
145  Arguments:
146  cfg (DictConfig): The entire configuration file as a DefaultMunch dictionary.
147 
148 
149  Returns:
150  None
151  """
152 
153  if cfg.data_augmentation and cfg.custom_data_augmentation:
154  raise ValueError("\nThe `data_augmentation` and `custom_data_augmentation` attributes "
155  "are mutually exclusive.\nPlease check your configuration file.")
156 
157  if cfg.data_augmentation:
158  data_aug = DefaultMunch.fromDict({})
159  # The name of the Model Zoo data augmentation function is 'data_augmentation'.
160  data_aug.function_name = "data_augmentation"
161  data_aug.config = deepcopy(cfg.data_augmentation)
162 
163  if cfg.custom_data_augmentation:
164  check_attributes(cfg.custom_data_augmentation,
165  expected=["function_name"],
166  optional=["config"],
167  section="custom_data_augmentation")
168  if cfg.custom_data_augmentation["function_name"] == "data_augmentation":
169  raise ValueError("\nThe function name `data_augmentation` is reserved.\n"
170  "Please use another name (attribute `function_name` in "
171  "the 'custom_data_augmentation' section).")
172 
173  data_aug = DefaultMunch.fromDict({})
174  data_aug.function_name = cfg.custom_data_augmentation.function_name
175  if cfg.custom_data_augmentation.config:
176  data_aug.config = deepcopy(cfg.custom_data_augmentation.config)
177 
178  cfg.data_augmentation = data_aug
